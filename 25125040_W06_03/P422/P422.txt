class Solution {
public:
    double myPow(double x, int n) {
        if (n == 0) return 1.0;
        if (n == 1) return x;
        if (n < 0) 
        {
            if (n == INT_MIN) return 1 / x * myPow(x, n + 1);
            return 1.0 / myPow(x, -n);
        }
        double ans = myPow(x, n / 2);
        if (n & 1) return ans * ans * x;
        return ans * ans;
    }
};
